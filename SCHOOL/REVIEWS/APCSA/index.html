Bubble Sort:
public static void bubbleSort (int[] list){
  boolean sorted = false;
  for (int top = list.length - 1; top > 0; top--){
    sorted = true;
    for (int i = 0; i < top; i++){
       if (list[i] > list[i+1]){
          sorted = false;
          int temp = list[i];
          list[i] = list[i+1];
          list[i+1] = temp;
      }
    }
  }
}
